type i64 = integer<64, true>;
fn do_stuff(x: i64);

fn main() {
  if (true) {
    while (false) {
      for (let x: i64 = 0; x < 40; x = x + 1) {
        do_stuff(x);
      }
    }
  } else {
    return;
  }
}
